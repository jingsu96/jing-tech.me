```js file=visually-hidden.js
import React from 'react';

const VisuallyHidden = React.forwardRef((props, ref) => {
  const { as: ElementType = 'div', children, ...restProps } = props;
  return (
    <ElementType
      style={{
        border: 0,
        clip: 'rect(0 0 0 0)',
        clipPath: 'inset(50%)',
        height: '1px',
        margin: '-1px',
        overflow: 'hidden',
        padding: 0,
        position: 'absolute',
        width: '1px',
        whiteSpace: 'nowrap',
      }}
      {...restProps}
      ref={ref}
    >
      {children}
    </ElementType>
  );
});

export default VisuallyHidden;
```
